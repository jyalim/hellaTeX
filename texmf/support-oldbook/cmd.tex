\newcommand\pref[1]{(\ref{#1})}
\newcommand\bref[1]{\[\ref{#1}\]}
\newcommand\cref[1]{\{\ref{#1}\}}

\newcommand{\aGraph}[2]{
  \begin{figure}[h]
    \centering
    \centerline{%
      \includegraphics[%
        width=1.45\linewidth,
        angle=#1,%
      ]{#2}%
    }
  \end{figure}
}

\newcommand{\nGraph}[2]{
  \begin{figure}[H]
    \centering
    \centerline{\includegraphics[width=#1\linewidth]{#2}}
  \end{figure}
}

\newcommand{\wGraph}[5]{
  \begin{figure}[H]
    \centering
    \captionsetup{width=#4\linewidth}
    \centerline{\includegraphics[width=#1\linewidth]{#2}}
    \caption{#5}
    \label{fig#3}
  \end{figure}
}

\newcommand{\fGraph}[5]{
  \begin{figure}[H]
    \centering
    \captionsetup{width=#4\linewidth}
    \centerline{\includegraphics[width=#1\linewidth]{#2}}
    \caption{#5}
    \label{#3}
  \end{figure}
}

\newcommand{\pGraph}[4]{
  \begin{figure}[H]
    \centering
    \centerline{\includegraphics[scale=#1]{#2}}
    \caption{#4}
    \label{fig:#3}
  \end{figure}
}

\newcommand{\pTable}[5]{
  \scalebox{1}{
     \centering
     \begin{tabular}{ #1 }  % ex: ccc
       \multicolumn{#2}{c}{#3} \\ % Number of Columns, Title
       \hline
       #4 \\ % Labeling of Columns
       \hline
       #5 % Table Data
       \hline
    \end{tabular}
  }
}
